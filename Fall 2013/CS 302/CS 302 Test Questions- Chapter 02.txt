4. Write the code for a recursive binary search function that looks for an integer in an array of sorted integers and returns -1 if not found. 

	int binSearch(int a[], int search, int high, int low)
	{
		int middle = (high - low) / 2;
		if(high >= low)
		{
			if(search < a[middle]) return binSearch(a,search,middle - 1, low);
			else if(search > a[middle]) return binSearch(a,search,high, middle+1);
			else return middle;
		}
		else return -1;
	}
